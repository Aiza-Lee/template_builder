% Unified LaTeX Template (corrected version)
% 本模板用于生成算法竞赛模板集，支持中英文混排和代码高亮
% 文档类选项可以包括：a4paper, 12pt, oneside/twoside 等
\documentclass[##CLASS_OPTIONS##]{ctexart} % ctexart支持中文排版

% Packages - 导入所需的LaTeX包
\usepackage[utf8]{inputenc}	% 输入编码支持
\usepackage{geometry}		% 页面布局设置
\usepackage{listings}		% 代码列表和高亮
\usepackage{xeCJK}			% 中文支持（XeLaTeX）
\usepackage{fontspec}		% 字体设置
\usepackage{xcolor}			% 颜色支持
\usepackage{hyperref}		% 超链接和PDF书签

% Geometry / Page - 页面布局设置
\geometry{
	landscape,					% landscape
	left=##LEFT_MARGIN##,		% 左边距值，如: 2cm
	right=##RIGHT_MARGIN##,		% 右边距值，如: 2cm
	top=##TOP_MARGIN##,			% 上边距值，如: 2.5cm
	bottom=##BOTTOM_MARGIN##	% 下边距值，如: 2.5cm
}

% Fonts - 字体设置
\setmainfont{##MAIN_FONT##}  % 西文主字体名称，如：Times New Roman
\setCJKmainfont[BoldFont={##CJK_BOLD_FONT##}, ItalicFont={##CJK_ITALIC_FONT##}]{##CJK_MAIN_FONT##}  % 中文主字体名称

% Hyperref configuration - PDF导航、书签和元数据设置
\hypersetup{
	colorlinks=##COLORLINKS##,					% 是否使用彩色链接: true/false
	linkcolor=##LINK_COLOR##,					% 内部链接颜色名称，如: blue
	filecolor=##FILE_COLOR##,					% 文件链接颜色名称，如: magenta
	urlcolor=##URL_COLOR##,						% URL链接颜色名称，如: cyan
	bookmarksnumbered=##NUMBERED_BOOKMARKS##,	% 书签是否包含编号: true/false
	bookmarksopen=##OPEN_BOOKMARKS##,			% 是否默认展开书签: true/false
	pdftitle={##TITLE##},						% PDF文档标题
	pdfauthor={##AUTHOR##},						% PDF文档作者
	pdfsubject={##SUBJECT##}					% PDF文档主题
}

% 颜色定义
\definecolor{section_color}{RGB}{##SECTION_COLOR_RGB##}					% 章节标题颜色RGB值，如：0,51,102
\definecolor{subsection_color}{RGB}{##SUBSECTION_COLOR_RGB##}			% 小节标题颜色RGB值
\definecolor{subsubsection_color}{RGB}{##SUBSUBSECTION_COLOR_RGB##}		% 子小节颜色RGB值
\definecolor{paragraph_color}{RGB}{##PARAGRAPH_COLOR_RGB##}				% 段落颜色RGB值
\definecolor{subparagraph_color}{RGB}{##SUBPARAGRAPH_COLOR_RGB##}		% 子段落颜色RGB值

% 代码列表样式设置
\definecolor{code_keyword}{RGB}{##CODE_KEYWORD_COLOR##}		% 关键词RGB颜色，如: 0,0,255
\definecolor{code_comment}{RGB}{##CODE_COMMENT_COLOR##}		% 注释RGB颜色，如: 0,128,0
\definecolor{code_string}{RGB}{##CODE_STRING_COLOR##}		% 字符串RGB颜色，如: 163,21,21
\definecolor{code_background}{RGB}{##CODE_BG_COLOR##}		% 背景RGB颜色，如: 248,248,248
	
% 代码字体设置
\lstset{basicstyle=\fontsize{##CODE_FONT_SIZE##}{##CODE_LINE_HEIGHT##}\ttfamily}  % 代码字体大小(pt)和行高
\lstset{
	language=C++,									% 代码语言: C++
	keywordstyle=\color{code_keyword}\bfseries,		% 关键词样式
	commentstyle=\color{code_comment}\itshape,		% 注释样式
	stringstyle=\color{code_string},				% 字符串样式
	numberstyle=\tiny\color{gray},					% 行号样式
	backgroundcolor=\color{code_background},		% 背景颜色
	frame=none,										% 边框样式: none, leftline, topline, bottomline, lines, single, shadowbox
	breaklines=##CODE_AUTO_BREAK_LINES##,			% 是否自动换行: true/false
	tabsize=##CODE_TAB_SIZE##,						% Tab宽度，如: 4
	numbers=##CODE_SHOW_NUMBERS##,					% 行号位置: left, right, none
	showstringspaces=false,							% 不显示字符串中的空格
	showspaces=false,								% 不显示空格
	showtabs=false,									% 不显示Tab
	aboveskip=##CODE_SPACE_BEFORE##,				% 代码块前间距
	belowskip=##CODE_SPACE_AFTER##					% 代码块后间距
}

% 段落和分栏设置
\setlength{\parskip}{##PARAGRAPH_SKIP##}			% 段落间距
\setlength{\parindent}{##PARAGRAPH_INDENT##}		% 段落缩进
\setlength{\columnsep}{##COLUMN_SEP##}				% 分栏间距

% 页面样式设置
\pagestyle{##PAGE_STYLE##}  % 页面风格: plain（只有页码）, headings（页眉带章节，页脚带页码）, empty（无页眉页脚）

% 文档标题元数据设置
\title{##TITLE##}			% 文档标题
\author{##AUTHOR##}			% 作者名称
\date{\today}

\begin{document}

	\maketitle			% 生成标题页
	##TITLE_NOTE##		% 标题页额外内容

	\newpage
	\tableofcontents	% 生成目录

	\newpage
	##CONTENT##			% 正文内容

\end{document}
