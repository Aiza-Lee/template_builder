\documentclass[10pt,landscape,twocolumn]{ctexart} % ctexart支持中文排版

\usepackage{geometry}		% 页面布局设置
\usepackage[outputdir=<<MINTED_OUTPUTDIR>>]{minted}			% 代码高亮显示
\usepackage{fontspec}		% 字体设置
\usepackage{xeCJK}			% 中文支持（XeLaTeX）
\usepackage{xcolor}			% 颜色支持
\usepackage{hyperref}		% 超链接和PDF书签

\setcounter{secnumdepth}{5}
\setcounter{tocdepth}{5}

% Geometry / Page - 页面布局设置
\geometry{
	##GEOMETRY_PAPER_SIZE##,				% 纸张大小，如: a4paper, letterpaper
	landscape,								% landscape
	left=##GEOMETRY_LEFT_MARGIN##,			% 左边距值，如: 2cm
	right=##GEOMETRY_RIGHT_MARGIN##,		% 右边距值，如: 2cm
	top=##GEOMETRY_TOP_MARGIN##,			% 上边距值，如: 2.5cm
	bottom=##GEOMETRY_BOTTOM_MARGIN##		% 下边距值，如: 2.5cm
}

\setlength{\columnsep}{##GEOMETRY_COLUMN_SEP##} % 双栏间距，如: 1cm

% Fonts - 字体设置
\setmainfont{##GLOBAL_MAIN_FONT##}  % 西文主字体名称，如：Times New Roman
\setCJKmainfont{##GLOBAL_CJK_MAIN_FONT##}[  % 中文主字体名称
	% BoldFont=##GLOBAL_CJK_MAIN_FONT## Bold,
	% ItalicFont=##GLOBAL_CJK_MAIN_FONT## Italic,
	AutoFakeBold=true,
	AutoFakeSlant=true
]

% Hyperref configuration - PDF导航、书签和元数据设置
\hypersetup{
	bookmarksdepth=5									% 最深目录层级
	colorlinks=##HYPERREF_ENABLE_COLORLINKS##,			% 是否使用彩色链接: true/false
	linkcolor=##HYPERREF_LINK_COLOR##,					% 内部链接颜色名称，如: blue
	filecolor=##HYPERREF_FILE_COLOR##,					% 文件链接颜色名称，如: magenta
	urlcolor=##HYPERREF_URL_COLOR##,					% URL链接颜色名称，如: cyan
	bookmarksnumbered=##HYPERREF_NUMBERED_BOOKMARKS##,	% 书签是否包含编号: true/false
	bookmarksopen=##HYPERREF_OPEN_BOOKMARKS##,			% 是否默认展开书签: true/false
	pdftitle={##TITLE_CONTENT##},						% PDF文档标题
	pdfauthor={##AUTHOR##},								% PDF文档作者
	pdfsubject={##SUBJECT##}							% PDF文档主题
}

\newfontfamily{\customcodefontfamily}{##CODE_FONT_FAMILY##}[		% Fira Code 字体设置
	NFSSFamily=CustomCodeFont,
	Contextuals={Alternate},  % 启用编程连字
	Ligatures={Common, Rare}, % 启用所有连字类型
	Kerning=On                % 开启字距调整
]

\setminted{
	style=##CODE_MINTED_STYLE##,
	fontfamily=CustomCodeFont,
	fontsize=\fontsize{##CODE_FONT_SIZE##}{##CODE_LINE_HEIGHT##}\selectfont,
	breaklines=true,
	tabsize=##CODE_TAB_SIZE##,
	linenos=true,
	numbersep=2pt,
	bgcolor=##CODE_BG_COLOR##,
	frame=none,
}

% 页面样式设置
\pagestyle{##GLOBAL_PAGE_STYLE##}  % 页面风格: plain（只有页码）, headings（页眉带章节，页脚带页码）, empty（无页眉页脚）

% 文档标题元数据设置
\title{\Large\bfseries{##TITLE_CONTENT##}}			% 文档标题
\author{##AUTHOR##}			% 作者名称
\date{\today}

\begin{document}

	\onecolumn			% 标题页和目录为单栏格式
	\maketitle			% 生成标题页
	\begin{center}
		##TITLE_NOTE##	% 标题页额外内容
	\end{center}

	\newpage
	\twocolumn			% 目录和正文为双栏格式
	\tableofcontents	% 生成目录

	\newpage
	<<CONTENT>>			% 正文内容

\end{document}
